(defn factorial [n]
  "computes n * (n-1) * ... * 1"
  (loop [result 1
         ivals (range 2 (+ n 1))]
    (if (seq ivals)
      (recur (*' result (first ivals))
             (rest ivals))
      result)))

(defn cumsum [values]
  "returns a vector containing the 
  cumulative sum of values in a collection."
  (reduce (fn [output v]
            (conj output 
                  (+ (peek output) v)))
          [] values))
